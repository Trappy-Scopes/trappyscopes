	def populate(partialdevice):
		if "lenses" in partialdevice:
			for l in partialdevice["lenses"]:
				if "zoomlens" in l:
					Optics.lenses.append(Lenses.Objective.ZoomLens())
				else:
					Optics.lenses.append(Lenses.Asphere(l))
